  v4l2-ctl -d $v --list-formats-ext 2>/dev/null | head -n 40
  echo
done

# video streaming via gstream
W=1280
H=720
FPS=30
PORT=8888

gst-launch-1.0 -e \
  v4l2src device=$DEV ! video/x-raw,format=NV12,width=$W,height=$H,framerate=${FPS}/1 ! \
  queue max-size-buffers=2 leaky=downstream ! \
  videoconvert ! video/x-raw,format=I420 ! \
  x264enc tune=zerolatency speed-preset=ultrafast key-int-max=$FPS bitrate=4000 ! \
  h264parse config-interval=1 ! mpegtsmux ! \
  tcpserversink host=127.0.0.1 port=$PORT sync=false
